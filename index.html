let coins = 0; let tapPower = 1; // Тап function tapCoin() { coins += tapPower; updateUI(); } // Магазин function buyUpgrade(type) { if (type === "tap1" && coins >= 100) { coins -= 100; tapPower += 1; } if (type === "tap5" && coins >= 500) { coins -= 500; tapPower += 5; } updateUI(); } // Обновление текста function updateUI() { document.getElementById("coins").innerText = coins; document.getElementById("power").innerText = tapPower; }
