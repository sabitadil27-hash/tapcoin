let coins = 100000; let tapPower = 1; let energy = 100; let friends = 0; let battlePassLevel = 1; // Тап function tapCoin() { if (energy > 0) { coins += tapPower; energy--; updateUI(); } } // Магазин function buyUpgrade(type) { if (type === "tap1" && coins >= 25000) { coins -= 25000; tapPower += 1; } if (type === "energy" && coins >= 80000) { coins -= 80000; energy += 100; } if (type === "vip" && coins >= 2000000) { coins -= 2000000; tapPower *= 2; } updateUI(); } // Друзья function inviteFriend() { friends++; coins += 100000; updateUI(); } // Battle Pass function levelUpPass() { if (coins >= 500000) { coins -= 500000; battlePassLevel++; } updateUI(); } // UI function updateUI() { document.getElementById("coins").innerText = coins; document.getElementById("energy").innerText = energy; document.getElementById("friends").innerText = friends; document.getElementById("pass").innerText = battlePassLevel; }
